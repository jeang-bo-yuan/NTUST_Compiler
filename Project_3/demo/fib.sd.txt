/* fib.sd
 *
 * This test program computes the Nth Fibonacci number
 * using recursion and iteration.
 */

// global variables
const int n = 8;

// iterative function
int itFibonacci (int n) {
  int Fn, FNminus1, temp;

  Fn = 1;
  FNminus1 = 1;
  while (n > 2) {
    temp = Fn;
    Fn = Fn + FNminus1;
    FNminus1 = temp;
    n = n - 1;
  }
  return Fn;
}

// recursive function
int recFibonacci (int n) {
  if (n <= 2)
    return 1;
  else
    return recFibonacci(n - 1) + recFibonacci(n - 2);
}

/* main program */
main()
{ 
  print "N: ";
  println n;
  println "Result of iterative computation:";
  println itFibonacci(n);
  println "Result of recursive computation:";
  println recFibonacci(n);
}
